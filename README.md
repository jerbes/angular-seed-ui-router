# install:

	git clone https://github.com/jerbes/angular-seed.git

	cd angular-seed/

	nvm use 0.10.32

	npm install

	npm start

# unit test:

	npm test

# functional test:

	npm start

	npm run update-webdriver

	npm run protractor
